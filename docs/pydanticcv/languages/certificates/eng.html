English language proficiency exam Pydantic models.

225        """Convert this record to the legacy 0-30 scoring scale.
226
227        Each section is converted independently using the reverse of the ETS

229        same new value, the returned section scores are the **highest** legacy
230        value that maps to each new score — not necessarily the original values.

232        Returns:
233            A new :class:`TOEFLiBTLegacy` instance with approximate legacy scores.
234        """
235        return TOEFLiBTLegacy(
236            DateTaken=self.DateTaken,

238            AwardingInstitution=self.AwardingInstitution,
239            Scores=TOEFLiBTLegacyScores(
240                Reading=READING_TO_LEGACY[self.Scores.Reading],
241                Listening=LISTENING_TO_LEGACY[self.Scores.Listening],
242                Speaking=SPEAKING_TO_LEGACY[self.Scores.Speaking],
243                Writing=WRITING_TO_LEGACY[self.Scores.Writing],
244            ),
245        )

125    """

131    @model_validator(mode="after")
132    def check_total_in_range(self) -> "TOEFLITP":
133        """Validate that the computed total falls within the expected range for the level.
134
135        Returns:
136            The validated model instance.
137
138        Raises:
139            ValueError: If the total score is outside the valid range for the chosen level.
140        """
141        lo, hi = _LEVEL_TOTAL_RANGE[self.Level]
142        total = self.Scores.Total
143        if not lo <= total <= hi:
144            raise ValueError(
145                f"TOEFL ITP {self.Level} total score must be {lo}-{hi}, got {total}"
146            )
147        return self
148
149    @computed_field
150    @property
151    def CEFRLevel(self) -> CEFRLiteral | None:
152        """Derive the approximate CEFR level from the Level 1 total score.
153
154        Returns:
155            CEFR level string for Level 1 records, or ``None`` for Level 2.
156        """
157        if self.Level != "Level 1":
158            return None
159        total = self.Scores.Total
160        for threshold, cefr in _ITP_LEVEL1_CEFR_THRESHOLDS:
161            if total >= threshold:
162                return cefr
163        return None
